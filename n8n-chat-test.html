Integration Code

Once your workflow is active, use this code in your website:

<script>
(function() {
    const CHAT_URL = 'https://czarkamxxx.app.n8n.cloud/webhook/ca436bba-e4cf-43d0-afa4-d80541d06722/chat';
    
    async function sendMessage(message) {
        try {
            const response = await fetch(CHAT_URL, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({
                    message: message,
                    sessionId: sessionStorage.getItem('chatSessionId') || 'session-' + Date.now()
                })
            });
            
            // Handle streaming response
            const reader = response.body.getReader();
            const decoder = new TextDecoder();
            let result = '';
            
            while (true) {
                const { value, done } = await reader.read();
                if (done) break;
                result += decoder.decode(value);
                // Update UI with streaming content
                updateChatUI(result);
            }
        } catch (error) {
            console.error('Chat error:', error);
            // Use fallback response
        }
    }
})();
</script>
